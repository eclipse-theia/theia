    Interface GenericCapabilitiesService

    Service to provide lists of available generic capabilities for selection in the UI. Aggregates capabilities from various sources (skills, MCP, functions, etc.) and provides change notifications when available capabilities change.

    interface GenericCapabilitiesService {
        onDidChangeAvailableCapabilities: Event<void>;
        getAvailableAgents(excludeAgentId?: string): GenericCapabilityItem[];
        getAvailableFunctions(): GenericCapabilityGroup[];
        getAvailableMCPFunctions(): Promise<GenericCapabilityGroup[]>;
        getAvailablePromptFragments(): GenericCapabilityItem[];
        getAvailableSkills(): GenericCapabilityItem[];
        getAvailableVariables(): GenericCapabilityItem[];
    }
